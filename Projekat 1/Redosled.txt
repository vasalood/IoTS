OVAKO SE U DOCKER POKRECE KONTEJNER MONGODB

docker run -d 
--name mongo-project-1 
-e MONGO_INITDB_ROOT_USERNAME=mongoadmin 
-e MONGO_INITDB_ROOT_PASSWORD=Vasamare123 
-p 27017:27017 
-v agriculture-data:/data/db mongo

KOPIRANJE DATASETA U KONTEJNER

docker cp ProjectData.csv mongo-project-1:/data/data.csv

IMPORT DATASETA U BAZU

docker exec -it mongo-project-1 mongoimport 
--username mongoadmin 
--password Vasamare123 
--authenticationDatabase admin 
--db agriculture_database 
--collection iot_data 
--type csv 
--headerline 
--file /data/data.csv
